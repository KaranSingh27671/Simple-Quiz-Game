#include <iostream>
using namespace std;

int main() {
    string questions[] = {
        "What is the capital of France?",
        "Which language is used for system programming?",
        "Who developed C++?"
    };
    
    string options[][4] = {
        {"1. London","2. Berlin","3. Paris","4. Rome"},
        {"1. Python","2. C","3. HTML","4. JavaScript"},
        {"1. Dennis Ritchie","2. Bjarne Stroustrup","3. James Gosling","4. Guido van Rossum"}
    };
    
    int answers[] = {3,2,2};
    int score = 0;
    
    for(int i=0;i<3;i++){
        cout << questions[i] << "\n";
        for(int j=0;j<4;j++) cout << options[i][j] << "\n";
        int choice;
        cout << "Your answer (1-4): ";
        cin >> choice;
        if(choice == answers[i]){
            cout << "Correct!\n";
            score++;
        } else cout << "Wrong!\n";
        cout << "\n";
    }
    
    cout << "Your total score: " << score << "/3\n";
    return 0;
}
